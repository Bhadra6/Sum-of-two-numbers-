# Sum-of-two-numbers-
public class SumOfTwoNumbers {
    public static void main(String[] args) {
        // Hardcoded numbers
        int num1 = 10;
        int num2 = 20;

        // Calculate sum
        int sum = num1 + num2;

        // Print the result
        System.out.println("The sum of " + num1 + " and " + num2 + " is: " + sum);
    }
}
